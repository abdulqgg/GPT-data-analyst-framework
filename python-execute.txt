import plotly.graph_objects as go

names = ['Helena', 'Holý', 'Hugh', "O'Reilly", 'Ladislav', 'Kovács', 'Luis', 'Rojas', 'Richard', 'Cunningham']
ages = [7, 7, 7, 7, 7, 7, 7, 7]
gpa = [49.62, 45.62, 45.62, 46.62, 47.62]

fig = go.Figure(data=[
    go.Bar(name='Age', x=names, y=ages),
    go.Bar(name='GPA', x=names, y=gpa)
])

fig.update_layout(title_text='Student Data', xaxis_title='Names', yaxis_title='Values')
fig.show()
